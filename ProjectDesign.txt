一、需求分析：
    1、项目目标：
        1、实现基于症状、疾病的问答对话系统。
        2、提供穴位图、经络图的查询功能。
        3、提供面向穴位的治疗方式建议。
    2、功能需求：
        1、用户注册：用户可以注册账号，并设置密码、手机号、性别等信息。
        2、用户登录：用户可以登录系统，并查看历史对话记录。
        3、问答对话：用户可以输入症状、疾病、问题，系统会自动匹配相应的答案。
        4、穴位图查询：用户可以输入症状、疾病、穴位名称，系统会返回相应的穴位。
图。
	5、病例生成：可以根据用户的历史记录生成病例。
    3、非功能需求：
        1、安全性：系统采用 HTTPS 加密传输，用户信息加密存储。
        2、可用性：系统具有良好的可用性，响应速度快，无明显卡顿。
        3、可扩展性：系统具有良好的可扩展性，可以方便地添加功能。
        4、可迁移性：系统具有良好的可迁移性，可以方便地迁移到其他服务器上。
        5、可维护性：系统具有良好的可维护性，代码结构清晰，易于修改。
二、数据库设计
1、表设计
    1、用户表
CREATE TABLE User (
    Id INT AUTO_INCREMENT PRIMARY KEY,
    UserName VARCHAR(50) NOT NULL UNIQUE,
    Password VARCHAR(255) NOT NULL,
    PhoneNumber VARCHAR(15),
    Gender ENUM('Male', 'Female', 'Other') NOT NULL  -- 性别字段，使用 ENUM 类型
);
    2、历史对话信息表
CREATE TABLE Historical_Dialogue_Info (
    DialogueId INT NOT NULL,
    SequenceId INT NOT NULL,
    Date DATETIME NOT NULL,
    Question TEXT NOT NULL,
    Responds TEXT NOT NULL,
    UserId INT,
    PRIMARY KEY (DialogueId, Date),  -- 将 DialogueId 和 SequenceId 作为复合主键
    FOREIGN KEY (UserId) REFERENCES User(Id) ON DELETE CASCADE
);
    3、穴位表
CREATE TABLE Acupoints (
    Id INT PRIMARY KEY,
    Name VARCHAR(255),
    AlternativeNames VARCHAR(255), -- You may want to split this into multiple columns if you need each alternative name as a separate field.
    LocationDescription TEXT,
    Image BLOB,
    Function TEXT,
    Technique TEXT,
    MeridianId INT,
    FOREIGN KEY (MeridianId) REFERENCES Meridians(Id)
);
    4、经络表
CREATE TABLE Meridians (
    Id INT PRIMARY KEY,
    Name VARCHAR(255),
    Description TEXT,
    Image BLOB
);
    5、病例表
CREATE TABLE MedicalCase (
    CaseId INT AUTO_INCREMENT PRIMARY KEY, -- 病例ID，自动增长主键
    PatientId INT NOT NULL, -- 患者ID，关联到患者表
    DiagnosisDate DATE NOT NULL, -- 诊断日期
    Symptoms TEXT, -- 症状描述
    Diagnosis TEXT, -- 诊断结果
    TreatmentPlan TEXT, -- 治疗计划
    FollowUpInstructions TEXT, -- 随访指示
    Notes TEXT, -- 其他备注
    FOREIGN KEY (PatientId) REFERENCES User(Id), -- 假设User表中存储了患者信息
);
